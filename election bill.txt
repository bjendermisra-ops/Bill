<script>
  // Electronics Items Data
  let electronicsItems = [
    { name: "स्मार्टफोन (5G)", qty: 1, price: 15999 },
    { name: "ब्लूटूथ हेडफोन", qty: 2, price: 1299 },
    { name: "पॉवर बैंक 10000mAh", qty: 1, price: 899 },
    { name: "USB Type-C केबल", qty: 3, price: 199 }
  ];

  const tableBody = document.getElementById("tableBody");
  const shopInput = document.getElementById("shopName");
  const customerInput = document.getElementById("customerName");
  const customerPhoneInput = document.getElementById("customerPhone");
  const billDateInput = document.getElementById("billDate");
  const billNumberInput = document.getElementById("billNumber");
  const warrantyInput = document.getElementById("warranty");
  const gstPercentInput = document.getElementById("gstPercent");
  const extraNoteInput = document.getElementById("extraNote");
  const discountAmountInput = document.getElementById("discountAmount");
  const discountPercentInput = document.getElementById("discountPercent");
  const previewDiv = document.getElementById("billPreviewContent");

  function renderTable() {
    tableBody.innerHTML = "";
    electronicsItems.forEach((item, idx) => {
      const row = document.createElement("tr");
      row.innerHTML = `
        <td><input type="text" class="item-name" value="${escapeHtml(item.name)}" data-idx="${idx}" style="min-width:180px;"></td>
        <td><input type="number" class="item-qty" value="${item.qty}" step="1" data-idx="${idx}" style="width:100px;"></td>
        <td><input type="number" class="item-price" value="${item.price}" step="100" data-idx="${idx}" style="width:120px;"></td>
        <td class="item-total">₹${(item.qty * item.price).toFixed(2)}</td>
        <td><button class="remove-row" data-idx="${idx}"><i class="fas fa-trash-alt"></i></button></td>
      `;
      tableBody.appendChild(row);
    });
    attachTableEvents();
    updateLiveBill();
  }

  function escapeHtml(str) { return str; }

  function attachTableEvents() {
    document.querySelectorAll(".item-name, .item-qty, .item-price").forEach(inp => {
      inp.removeEventListener("input", handleItemChange);
      inp.addEventListener("input", handleItemChange);
    });
    document.querySelectorAll(".remove-row").forEach(btn => {
      btn.removeEventListener("click", handleRemoveItem);
      btn.addEventListener("click", handleRemoveItem);
    });
  }

  function handleItemChange(e) {
    const idx = e.target.getAttribute("data-idx");
    if (idx !== null) {
      const nameInput = document.querySelector(`.item-name[data-idx='${idx}']`);
      const qtyInput = document.querySelector(`.item-qty[data-idx='${idx}']`);
      const priceInput = document.querySelector(`.item-price[data-idx='${idx}']`);
      if (nameInput && qtyInput && priceInput) {
        electronicsItems[idx].name = nameInput.value;
        electronicsItems[idx].qty = parseFloat(qtyInput.value) || 0;
        electronicsItems[idx].price = parseFloat(priceInput.value) || 0;
      }
    }
    updateTotalsInTable();
    updateLiveBill();
  }

  function handleRemoveItem(e) {
    const idx = e.currentTarget.getAttribute("data-idx");
    if (idx !== null) {
      electronicsItems.splice(parseInt(idx), 1);
      renderTable();
    }
  }

  function updateTotalsInTable() {
    const rows = tableBody.querySelectorAll("tr");
    rows.forEach((row, idx) => {
      if (electronicsItems[idx]) {
        const totalCell = row.querySelector(".item-total");
        const totalAmt = electronicsItems[idx].qty * electronicsItems[idx].price;
        if (totalCell) totalCell.innerText = `₹${totalAmt.toFixed(2)}`;
      }
    });
  }

  function getBillData() {
    let items = electronicsItems.map(i => ({
      name: i.name,
      quantity: i.qty,
      price: i.price,
      total: i.qty * i.price
    }));
    let subtotal = items.reduce((sum, i) => sum + i.total, 0);
    let discountAmt = parseFloat(discountAmountInput.value) || 0;
    let discountPercent = parseFloat(discountPercentInput.value) || 0;
    let discountFromPercent = subtotal * (discountPercent / 100);
    let totalDiscount = discountAmt + discountFromPercent;
    let afterDiscount = subtotal - totalDiscount;
    let gstPercent = parseFloat(gstPercentInput.value) || 0;
    let gstAmount = afterDiscount * (gstPercent / 100);
    let grandTotal = afterDiscount + gstAmount;
    return { items, subtotal, totalDiscount, afterDiscount, gstPercent, gstAmount, grandTotal };
  }

  function updateLiveBill() {
    const shop = shopInput.value || "इलेक्ट्रॉनिक स्टोर";
