<script>
  // Kirana Items Data
  let kiranaItems = [
    { name: "आटा (गेहूं) 5kg", qty: 2, price: 220 },
    { name: "बासमती चावल 5kg", qty: 1, price: 350 },
    { name: "चीनी 2kg", qty: 1, price: 80 },
    { name: "मूंग दाल 1kg", qty: 1, price: 120 }
  ];

  const tableBody = document.getElementById("tableBody");
  const businessInput = document.getElementById("businessName");
  const customerInput = document.getElementById("customerName");
  const billDateInput = document.getElementById("billDate");
  const billNumberInput = document.getElementById("billNumber");
  const gstPercentInput = document.getElementById("gstPercent");
  const extraNoteInput = document.getElementById("extraNote");
  const discountInput = document.getElementById("discountAmount");
  const previewDiv = document.getElementById("billPreviewContent");

  function renderTable() {
    tableBody.innerHTML = "";
    kiranaItems.forEach((item, idx) => {
      const row = document.createElement("tr");
      row.innerHTML = `
        <td><input type="text" class="item-name" value="${escapeHtml(item.name)}" data-idx="${idx}" style="min-width:150px;"></td>
        <td><input type="number" class="item-qty" value="${item.qty}" step="0.5" data-idx="${idx}" style="width:100px;"></td>
        <td><input type="number" class="item-price" value="${item.price}" step="5" data-idx="${idx}" style="width:110px;"></td>
        <td class="item-total">₹${(item.qty * item.price).toFixed(2)}</td>
        <td><button class="remove-row" data-idx="${idx}"><i class="fas fa-trash-alt"></i> हटाएं</button></td>
      `;
      tableBody.appendChild(row);
    });
    attachTableEvents();
    updateLiveBill();
  }

  function escapeHtml(str) { return str; }

  function attachTableEvents() {
    document.querySelectorAll(".item-name").forEach(inp => {
      inp.removeEventListener("input", handleItemChange);
      inp.addEventListener("input", handleItemChange);
    });
    document.querySelectorAll(".item-qty").forEach(inp => {
      inp.removeEventListener("input", handleItemChange);
      inp.addEventListener("input", handleItemChange);
    });
    document.querySelectorAll(".item-price").forEach(inp => {
      inp.removeEventListener("input", handleItemChange);
      inp.addEventListener("input", handleItemChange);
    });
    document.querySelectorAll(".remove-row").forEach(btn => {
      btn.removeEventListener("click", handleRemoveItem);
      btn.addEventListener("click", handleRemoveItem);
    });
  }

  function handleItemChange(e) {
    const idx = e.target.getAttribute("data-idx");
    if (idx !== null) {
      const nameInput = document.querySelector(`.item-name[data-idx='${idx}']`);
      const qtyInput = document.querySelector(`.item-qty[data-idx='${idx}']`);
      const priceInput = document.querySelector(`.item-price[data-idx='${idx}']`);
      if (nameInput && qtyInput && priceInput) {
        kiranaItems[idx].name = nameInput.value;
        kiranaItems[idx].qty = parseFloat(qtyInput.value) || 0;
        kiranaItems[idx].price = parseFloat(priceInput.value) || 0;
      }
    }
    updateTotalsInTable();
    updateLiveBill();
  }

  function handleRemoveItem(e) {
    const idx = e.currentTarget.getAttribute("data-idx");
    if (idx !== null) {
      kiranaItems.splice(parseInt(idx), 1);
      renderTable();
    }
  }

  function updateTotalsInTable() {
    const rows = tableBody.querySelectorAll("tr");
    rows.forEach((row, idx) => {
      if (kiranaItems[idx]) {
        const totalCell = row.querySelector(".item-total");
        const totalAmt = kiranaItems[idx].qty * kiranaItems[idx].price;
        if (totalCell) totalCell.innerText = `₹${totalAmt.toFixed(2)}`;
      }
    });
  }

  function getBillData() {
    let items = kiranaItems.map(i => ({
      name: i.name,
      quantity: i.qty,
      price: i.price,
      total: i.qty * i.price
    }));
    let subtotal = items.reduce((sum, i) => sum + i.total, 0);
    let discount = parseFloat(discountInput.value) || 0;
    let afterDiscount = subtotal - discount;
    let gstPercent = parseFloat(gstPercentInput.value) || 0;
    let gstAmount = afterDiscount * (gstPercent / 100);
    let grandTotal = afterDiscount + gstAmount;
    return { items, subtotal, discount, afterDiscount, gstPercent, gstAmount, grandTotal };
  }

  function updateLiveBill() {
    const business = businessInput.value || "किराना स्टोर";
    const customer = customerInput.value || "ग्राहक";
    const date = billDateInput.value || new Date().toISOString().slice(0,10);
    const billNo = billNumberInput.value || "KIR-INV";
    const extraNote = extraNoteInput.value || "";

  async function captureBill() {

    const originalBg = element.style.backgroundColor;

    const canvas = await html2canvas(element, { scale: 2.5, backgroundColor: "#ffffff", logging: false });
    element.style.backgroundColor = originalBg;
